    <script>

        // Game state

        let gameMode = null;

        let playerScore = 0;

        let computerScore = 0;

        let player1Score = 0;

        let player2Score = 0;

        let playerChoice = null;

        let computerChoice = null;

        let player1Choice = null;

        let player2Choice = null;

        let waitingForOpponent = true;


        // DOM elements

        const modeSelection = document.getElementById('mode-selection');

        const singlePlayerGame = document.getElementById('single-player-game');

        const multiplayerGame = document.getElementById('multiplayer-game');

        const waitingRoom = document.getElementById('waiting-room');

        const multiplayerGameplay = document.getElementById('multiplayer-gameplay');

        const gameLink = document.getElementById('game-link');


        // Start single player game

        function startSinglePlayer() {

            gameMode = 'single';

            modeSelection.classList.add('hidden');

            singlePlayerGame.classList.remove('hidden');

            resetSinglePlayerGame();

        }


        // Start multiplayer game

        function startMultiplayer() {

            gameMode = 'multiplayer';

            modeSelection.classList.add('hidden');

            multiplayerGame.classList.remove('hidden');

            

            // Simulate finding an opponent after 3 seconds

            setTimeout(() => {

                waitingRoom.classList.add('hidden');

                multiplayerGameplay.classList.remove('hidden');

                waitingForOpponent = false;

            }, 3000);

        }


        // Back to mode selection

        function backToModeSelection() {

            gameMode = null;

            singlePlayerGame.classList.add('hidden');

            multiplayerGame.classList.add('hidden');

            waitingRoom.classList.remove('hidden');

            multiplayerGameplay.classList.add('hidden');

            modeSelection.classList.remove('hidden');

        }


        // Make choice in single player

        function makeChoice(choice) {

            playerChoice = choice;

            

            // Update player choice icon

            const playerChoiceIcon = document.getElementById('player-choice-icon');

            playerChoiceIcon.className = `fas fa-hand-${choice} text-4xl text-lettuce`;

            

            // Computer makes random choice

            const choices = ['rock', 'paper', 'scissors'];

            computerChoice = choices[Math.floor(Math.random() * choices.length)];

            

            // Update computer choice icon

            const computerChoiceIcon = document.getElementById('computer-choice-icon');

            computerChoiceIcon.className = `fas fa-hand-${computerChoice} text-4xl text-gray-700`;

            

            // Determine winner

            const result = determineWinner(playerChoice, computerChoice);

            

            // Update result display

            const resultDisplay = document.getElementById('result-display');

            resultDisplay.textContent = result.message;

            

            // Update scores

            if (result.winner === 'player') {

                playerScore++;

                document.getElementById('player-score').textContent = playerScore;

            } else if (result.winner === 'computer') {

                computerScore++;

                document.getElementById('computer-score').textContent = computerScore;

            }

        }


        // Make choice in multiplayer

        function makeMultiplayerChoice(choice) {

            if (waitingForOpponent) return;

            

            player1Choice = choice;

            

            // Update player choice icon

            const player1ChoiceIcon = document.getElementById('player1-choice-icon');

            player1ChoiceIcon.className = `fas fa-hand-${choice} text-4xl text-lettuce`;

            

            // Simulate opponent making a choice after 1 second

            setTimeout(() => {

                const choices = ['rock', 'paper', 'scissors'];

                player2Choice = choices[Math.floor(Math.random() * choices.length)];

                

                // Update opponent choice icon

                const player2ChoiceIcon = document.getElementById('player2-choice-icon');

                player2ChoiceIcon.className = `fas fa-hand-${player2Choice} text-4xl text-gray-700`;

                

                // Determine winner

                const result = determineWinner(player1Choice, player2Choice);

                

                // Update result display

                const multiplayerResult = document.getElementById('multiplayer-result');

                multiplayerResult.textContent = result.message;

                

                // Update scores

                if (result.winner === 'player1') {

                    player1Score++;

                    document.getElementById('player1-score').textContent = player1Score;

                } else if (result.winner === 'player2') {

                    player2Score++;

                    document.getElementById('player2-score').textContent = player2Score;

                }

                

                // Reset choices after 2 seconds

                setTimeout(() => {

                    player1Choice = null;

                    player2Choice = null;

                    player1ChoiceIcon.className = 'fas fa-question text-4xl text-gray-400';

                    player2ChoiceIcon.className = 'fas fa-question text-4xl text-gray-400';

                    multiplayerResult.textContent = 'VS';

                }, 2000);

            }, 1000);

        }


        // Determine winner

        function determineWinner(choice1, choice2) {

            if (choice1 === choice2) {

                return { winner: null, message: 'Draw!' };

            }

            

            if (

                (choice1 === 'rock' && choice2 === 'scissors') ||

                (choice1 === 'paper' && choice2 === 'rock') ||

                (choice1 === 'scissors' && choice2 === 'paper')

            ) {

                if (gameMode === 'single') {

                    return { winner: 'player', message: 'You win!' };

                } else {

                    return { winner: 'player1', message: 'You win!' };

                }

            } else {

                if (gameMode === 'single') {

                    return { winner: 'computer', message: 'Computer wins!' };

                } else {

                    return { winner: 'player2', message: 'Opponent wins!' };

                }

            }

        }


        // Reset single player game

        function resetSinglePlayerGame() {

            playerScore = 0;

            computerScore = 0;

            playerChoice = null;

            computerChoice = null;

            

            document.getElementById('player-score').textContent = '0';

            document.getElementById('computer-score').textContent = '0';

            document.getElementById('player-choice-icon').className = 'fas fa-question text-4xl text-gray-400';

            document.getElementById('computer-choice-icon').className = 'fas fa-question text-4xl text-gray-400';

            document.getElementById('result-display').textContent = 'VS';

        }


        // Copy game link

        function copyGameLink() {

            navigator.clipboard.writeText(gameLink.textContent);

            

            // Show copied feedback

            const copyBtn = document.querySelector('[onclick="copyGameLink()"]');

            const originalText = copyBtn.innerHTML;

            copyBtn.innerHTML = '<i class="fas fa-check mr-2"></i> Copied!';

            

            setTimeout(() => {

                copyBtn.innerHTML = originalText;

            }, 2000);

        }
